useNuiEvent
Default value : <T = any>( action: string, handler: (data: T) => void ) => { const savedHandler: MutableRefObject<NuiHandlerSignature<T>> = useRef(noop); // When handler value changes set mutable ref to handler val useEffect(() => { savedHandler.current = handler; }, [handler]); useEffect(() => { const eventListener = (event: MessageEvent<NuiMessageData<T>>) => { const { action: eventAction, data } = event.data; if (savedHandler.current) { if (eventAction === action) { savedHandler.current(data); } } }; window.addEventListener("message", eventListener); // Remove Event Listener on component cleanup return () => window.removeEventListener("message", eventListener); }, [action]); }

A hook that manage events listeners for receiving data from the client scripts
