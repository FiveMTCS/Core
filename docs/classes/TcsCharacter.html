class TcsCharacter {
	private owner: TcsPlayer;

	private characterId: string;
	private informations: CharacterInformations;
	private skin: Object;

	constructor(
		owner: TcsPlayer,
		characterId: string,
		informations: CharacterInformations,
		skin: Object
	) {
		this.owner = owner;
		this.characterId = characterId;
		this.informations = informations;
		this.skin = skin;
	}

	getInformations(): CharacterInformations {
		return this.informations;
	}

	async setInformations(informations: CharacterInformations) {
		TCS.debug(`Registering new informations for character ${this.characterId}`);
		await TCS.database?.database().update(
			PLAYER_MODULE_CHARACTERS_TABLE,
			{
				id: this.characterId,
			},
			{ informations }
		);
		this.informations = informations;
	}

	getSkin(): Object {
		return this.skin;
	}

	async setSkin(skin: Object) {
		TCS.debug(`Registering new skin for character ${this.characterId}`);
		await TCS.database?.database().update(
			PLAYER_MODULE_CHARACTERS_TABLE,
			{
				id: this.characterId,
			},
			{ skin }
		);
		this.skin = skin;
	}

	getOwner(): TcsPlayer {
		return this.owner;
	}
}
